<script>


Papa.parse("aerofit_treadmill.csv",{

    download:true,

    header:true,


    complete:function(results){


        let data = results.data;


        populateFilters(data);


        document.getElementById("productFilter")

        .addEventListener("change",()=>updateDashboard(data));


        document.getElementById("genderFilter")

        .addEventListener("change",()=>updateDashboard(data));


        updateDashboard(data);

    }

});


function populateFilters(data){


    let products=[...new Set(data.map(x=>x.Product))];

    let genders=[...new Set(data.map(x=>x.Gender))];


    products.forEach(p=>{

        productFilter.innerHTML+=

        `<option value="${p}">${p}</option>`;

    });


    genders.forEach(g=>{

        genderFilter.innerHTML+=

        `<option value="${g}">${g}</option>`;

    });

}


function updateDashboard(data){


    let product=document.getElementById("productFilter").value;

    let gender=document.getElementById("genderFilter").value;


    let filtered=data.filter(row=>{


        let p=(product==="All" || row.Product===product);

        let g=(gender==="All" || row.Gender===gender);


        return p && g;

    });


    let incomeAvg =

        filtered.reduce((a,b)=>a+Number(b.Income),0)

        / filtered.length;


    let milesAvg =

        filtered.reduce((a,b)=>a+Number(b.Miles),0)

        / filtered.length;


    let fitnessAvg =

        filtered.reduce((a,b)=>a+Number(b.Fitness),0)

        / filtered.length;


    document.getElementById("customers")

        .innerHTML=filtered.length;


    document.getElementById("income")

        .innerHTML="₹"+incomeAvg.toFixed(0);


    document.getElementById("miles")

        .innerHTML=milesAvg.toFixed(0);


    document.getElementById("fitness")

        .innerHTML=fitnessAvg.toFixed(1);


    createPie(filtered);

    createAge(filtered);

    createScatter(filtered);

    createBox(filtered);

}


function createPie(data){


    let counts={};


    data.forEach(r=>{

        counts[r.Product]=(counts[r.Product]||0)+1;

    });


    Plotly.newPlot("pie",[{

        labels:Object.keys(counts),

        values:Object.values(counts),

        type:"pie"

    }],{

        title:"Product Market Share"

    });

}


function createAge(data){


    Plotly.newPlot("age",[{

        x:data.map(x=>x.Age),

        type:"histogram"

    }],{

        title:"Age Distribution"

    });

}


function createScatter(data){


    Plotly.newPlot("scatter",[{

        x:data.map(x=>x.Usage),

        y:data.map(x=>x.Miles),

        mode:"markers",

        text:data.map(x=>x.Product)

    }],{

        title:"Usage vs Miles",

        xaxis:{title:"Usage"},

        yaxis:{title:"Miles"}

    });

}


function createBox(data){


    let products=[...new Set(data.map(x=>x.Product))];


    let traces=products.map(product=>{


        return{

            y:data

            .filter(x=>x.Product===product)

            .map(x=>x.Income),


            type:"box",

            name:product

        };

    });


    Plotly.newPlot("box",traces,{

        title:"Income by Product"

    });

}
